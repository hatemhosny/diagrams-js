Example 1: Grouped Workers on AWS

TypeScript Code
import { Diagram } from "ts-diagrams";
import { EC2 } from "ts-diagrams/aws/compute";
import { RDS } from "ts-diagrams/aws/database";
import { ALB } from "ts-diagrams/aws/network";

const diagram = new Diagram("Grouped Workers", {
  direction: "TB",
});

const lb = diagram.add(new ALB("lb"));
const workers = [
  diagram.add(new EC2("worker1")),
  diagram.add(new EC2("worker2")),
  diagram.add(new EC2("worker3")),
  diagram.add(new EC2("worker4")),
  diagram.add(new EC2("worker5")),
];
const events = diagram.add(new RDS("events"));

lb.to(workers);
workers.forEach(w => w.to(events));

const svg = await diagram.render();

Example 2: Clustered Web Services

TypeScript Code
import { Diagram } from "ts-diagrams";
import { ECS } from "ts-diagrams/aws/compute";
import { RDS, ElastiCache } from "ts-diagrams/aws/database";
import { ALB, Route53 } from "ts-diagrams/aws/network";

const diagram = new Diagram("Clustered Web Services");

const dns = diagram.add(new Route53("dns"));
const lb = diagram.add(new ALB("lb"));

const svcCluster = diagram.cluster("Services");
const svcGroup = [
  svcCluster.add(new ECS("web1")),
  svcCluster.add(new ECS("web2")),
  svcCluster.add(new ECS("web3")),
];
lb.to(svcGroup);

const dbCluster = diagram.cluster("DB Cluster");
const dbPrimary = dbCluster.add(new RDS("userdb"));
const dbReplica = dbCluster.add(new RDS("userdb ro"));
dbPrimary.with(dbReplica);
svcGroup.forEach(s => s.to(dbPrimary));

const memcached = diagram.add(new ElastiCache("memcached"));
svcGroup.forEach(s => s.to(memcached));

dns.to(lb);

const svg = await diagram.render();

Example 3: Event Processing on AWS

TypeScript Code
import { Diagram } from "ts-diagrams";
import { ECS, EKS, Lambda } from "ts-diagrams/aws/compute";
import { Redshift } from "ts-diagrams/aws/database";
import { SQS } from "ts-diagrams/aws/integration";
import { S3 } from "ts-diagrams/aws/storage";

const diagram = new Diagram("Event Processing");

const source = diagram.add(new EKS("k8s source"));
const store = diagram.add(new S3("events store"));
const dw = diagram.add(new Redshift("analytics"));

const flowsCluster = diagram.cluster("Event Flows");

const workersCluster = flowsCluster.cluster("Event Workers");
const workerNodes = [
  workersCluster.add(new ECS("worker1")),
  workersCluster.add(new ECS("worker2")),
  workersCluster.add(new ECS("worker3")),
];
source.to(workerNodes);

const queue = flowsCluster.add(new SQS("event queue"));
workerNodes.forEach(w => w.to(queue));

const procCluster = flowsCluster.cluster("Processing");
const handlers = [
  procCluster.add(new Lambda("proc1")),
  procCluster.add(new Lambda("proc2")),
  procCluster.add(new Lambda("proc3")),
];
queue.to(handlers);
handlers.forEach(h => {
  h.to(store);
  h.to(dw);
});

const svg = await diagram.render();

Example 4: Message Collecting System on GCP

TypeScript Code
import { Diagram } from "ts-diagrams";
import { BigQuery, Dataflow, PubSub } from "ts-diagrams/gcp/analytics";
import { AppEngine, Functions } from "ts-diagrams/gcp/compute";
import { BigTable } from "ts-diagrams/gcp/database";
import { IotCore } from "ts-diagrams/gcp/iot";
import { GCS } from "ts-diagrams/gcp/storage";

const diagram = new Diagram("Message Collecting");

const pubsub = diagram.add(new PubSub("pubsub"));

const sourceCluster = diagram.cluster("Source of Data");
const sources = [
  sourceCluster.add(new IotCore("core1")),
  sourceCluster.add(new IotCore("core2")),
  sourceCluster.add(new IotCore("core3")),
];
sources.forEach(s => s.to(pubsub));

const targetsCluster = diagram.cluster("Targets");
const flowCluster = targetsCluster.cluster("Data Flow");
const dataflow = flowCluster.add(new Dataflow("data flow"));
pubsub.to(dataflow);

const lakeCluster = targetsCluster.cluster("Data Lake");
const bq = lakeCluster.add(new BigQuery("bq"));
const gcs = lakeCluster.add(new GCS("storage"));
dataflow.to([bq, gcs]);

const eventCluster = targetsCluster.cluster("Event Driven");

const procCluster = eventCluster.cluster("Processing");
const engine = procCluster.add(new AppEngine("engine"));
const bigtable = procCluster.add(new BigTable("bigtable"));
dataflow.to(engine).to(bigtable);

const serverlessCluster = eventCluster.cluster("Serverless");
const func = serverlessCluster.add(new Functions("func"));
const appengine = serverlessCluster.add(new AppEngine("appengine"));
dataflow.to(func).to(appengine);

const svg = await diagram.render();

Example 5: Exposed Pod with 3 Replicas on Kubernetes

TypeScript Code
import { Diagram } from "ts-diagrams";
import { HPA } from "ts-diagrams/k8s/clusterconfig";
import { Deployment, Pod, ReplicaSet } from "ts-diagrams/k8s/compute";
import { Ingress, Service } from "ts-diagrams/k8s/network";

const diagram = new Diagram("Exposed Pod with 3 Replicas");

const ingress = diagram.add(new Ingress("domain.com"));
const svc = diagram.add(new Service("svc"));
const pods = [
  diagram.add(new Pod("pod1")),
  diagram.add(new Pod("pod2")),
  diagram.add(new Pod("pod3")),
];
const rs = diagram.add(new ReplicaSet("rs"));
const dp = diagram.add(new Deployment("dp"));
const hpa = diagram.add(new HPA("hpa"));

// Forward: ingress >> svc >> pods
// Reverse: pods << rs << dp << hpa
ingress.to(svc);
svc.to(pods);
pods.forEach(p => p.from(rs));
rs.from(dp);
dp.from(hpa);

const svg = await diagram.render();

Example 6: Custom Nodes with Remote Icons

TypeScript Code
import { Diagram, Custom, Cluster } from "ts-diagrams";

const diagram = new Diagram("Custom with remote icons", {
  direction: "LR"
});

// Use remote icons from GitHub (PNG images)
const diagrams = diagram.add(
  new Custom("Diagrams", "https://raw.githubusercontent.com/mingrammer/diagrams/master/assets/img/diagrams.png")
);

const providersCluster = diagram.cluster("Some Providers");

const openstack = providersCluster.add(
  new Custom("OpenStack", "https://raw.githubusercontent.com/mingrammer/diagrams/master/resources/openstack/openstack.png")
);

const elastic = providersCluster.add(
  new Custom("Elastic", "https://raw.githubusercontent.com/mingrammer/diagrams/master/resources/elastic/saas/elastic.png")
);

diagrams.to(openstack);
diagrams.to(elastic);

const svg = await diagram.render();
