Example 1: Grouped Workers on AWS

TypeScript Code
import { Diagram } from "diagrams-ts";
import { EC2 } from "diagrams-ts/providers/aws/compute";
import { RDS } from "diagrams-ts/providers/aws/database";
import { ALB } from "diagrams-ts/providers/aws/network";

const diagram = new Diagram("Grouped Workers", {
  direction: "TB",
  outformat: "svg",
});

const lb = diagram.add(new ALB("lb"));
const workers = [
  diagram.add(new EC2("worker1")),
  diagram.add(new EC2("worker2")),
  diagram.add(new EC2("worker3")),
  diagram.add(new EC2("worker4")),
  diagram.add(new EC2("worker5")),
];
const events = diagram.add(new RDS("events"));

lb.to(workers);
workers.forEach(w => w.to(events));

const svg = await diagram.render();

Example 2: Clustered Web Services

TypeScript Code
import { Diagram, Cluster } from "diagrams-ts";
import { ECS } from "diagrams-ts/providers/aws/compute";
import { RDS, ElastiCache } from "diagrams-ts/providers/aws/database";
import { ALB, Route53 } from "diagrams-ts/providers/aws/network";

const diagram = new Diagram("Clustered Web Services");

const dns = diagram.add(new Route53("dns"));
const lb = diagram.add(new ALB("lb"));

const memcached = diagram.add(new ElastiCache("memcached"));

diagram.cluster("Services", (svc) => {
  const svcGroup = [
    svc.add(new ECS("web1")),
    svc.add(new ECS("web2")),
    svc.add(new ECS("web3")),
  ];
  lb.to(svcGroup);
  svcGroup.forEach(s => s.to(memcached));
  
  svc.cluster("DB Cluster", (db) => {
    const dbPrimary = db.add(new RDS("userdb"));
    svcGroup.forEach(s => s.to(dbPrimary));
  });
});

dns.to(lb);

const svg = await diagram.render();

Example 3: Event Processing on AWS

TypeScript Code
import { Diagram, Cluster } from "diagrams-ts";
import { ECS, EKS, Lambda } from "diagrams-ts/providers/aws/compute";
import { Redshift } from "diagrams-ts/providers/aws/database";
import { SQS } from "diagrams-ts/providers/aws/integration";
import { S3 } from "diagrams-ts/providers/aws/storage";

const diagram = new Diagram("Event Processing");

const source = diagram.add(new EKS("k8s source"));
const store = diagram.add(new S3("events store"));
const dw = diagram.add(new Redshift("analytics"));

diagram.cluster("Event Flows", (flows) => {
  flows.cluster("Event Workers", (workers) => {
    const workerNodes = [
      workers.add(new ECS("worker1")),
      workers.add(new ECS("worker2")),
      workers.add(new ECS("worker3")),
    ];
    source.to(workerNodes);
    
    const queue = flows.add(new SQS("event queue"));
    workerNodes.forEach(w => w.to(queue));
    
    flows.cluster("Processing", (proc) => {
      const handlers = [
        proc.add(new Lambda("proc1")),
        proc.add(new Lambda("proc2")),
        proc.add(new Lambda("proc3")),
      ];
      queue.to(handlers);
      handlers.forEach(h => {
        h.to(store);
        h.to(dw);
      });
    });
  });
});

const svg = await diagram.render();

Example 4: Message Collecting System on GCP

TypeScript Code
import { Diagram, Cluster } from "diagrams-ts";
import { BigQuery, Dataflow, PubSub } from "diagrams-ts/providers/gcp/analytics";
import { AppEngine, Functions } from "diagrams-ts/providers/gcp/compute";
import { BigTable } from "diagrams-ts/providers/gcp/database";
import { IotCore } from "diagrams-ts/providers/gcp/iot";
import { GCS } from "diagrams-ts/providers/gcp/storage";

const diagram = new Diagram("Message Collecting");

const pubsub = diagram.add(new PubSub("pubsub"));

diagram.cluster("Source of Data", (source) => {
  const sources = [
    source.add(new IotCore("core1")),
    source.add(new IotCore("core2")),
    source.add(new IotCore("core3")),
  ];
  sources.forEach(s => s.to(pubsub));
});

diagram.cluster("Targets", (targets) => {
  targets.cluster("Data Flow", (flow) => {
    const dataflow = flow.add(new Dataflow("data flow"));
    pubsub.to(dataflow);
    
    targets.cluster("Data Lake", (lake) => {
      const bq = lake.add(new BigQuery("bq"));
      const gcs = lake.add(new GCS("storage"));
      dataflow.to([bq, gcs]);
    });
    
    targets.cluster("Event Driven", (event) => {
      event.cluster("Processing", (proc) => {
        const engine = proc.add(new AppEngine("engine"));
        const bigtable = proc.add(new BigTable("bigtable"));
        dataflow.to(engine).to(bigtable);
      });
      
      event.cluster("Serverless", (serverless) => {
        const func = serverless.add(new Functions("func"));
        const appengine = serverless.add(new AppEngine("appengine"));
        dataflow.to(func).to(appengine);
      });
    });
  });
});

const svg = await diagram.render();

Example 5: Exposed Pod with 3 Replicas on Kubernetes

TypeScript Code
import { Diagram } from "diagrams-ts";
import { HPA } from "diagrams-ts/providers/k8s/clusterconfig";
import { Deployment, Pod, ReplicaSet } from "diagrams-ts/providers/k8s/compute";
import { Ingress, Service } from "diagrams-ts/providers/k8s/network";

const diagram = new Diagram("Exposed Pod with 3 Replicas");

const ingress = diagram.add(new Ingress("domain.com"));
const svc = diagram.add(new Service("svc"));
const pods = [
  diagram.add(new Pod("pod1")),
  diagram.add(new Pod("pod2")),
  diagram.add(new Pod("pod3")),
];
const rs = diagram.add(new ReplicaSet("rs"));
const dp = diagram.add(new Deployment("dp"));
const hpa = diagram.add(new HPA("hpa"));

// Forward: ingress >> svc >> pods
// Reverse: pods << rs << dp << hpa
ingress.to(svc);
svc.to(pods);
pods.forEach(p => p.from(rs));
rs.from(dp);
dp.from(hpa);

const svg = await diagram.render();
